This is the code for a guessing game. The program chooses a number between 0 and 10.

The user then guesses a number, and the program tells the user if they picked right. This is a secret program.

import random


# create a variable named rounds that stores how many rounds have passed
rounds = 0
while rounds != 3:
    # we create a variable which stores the number chosen by the program
    chosen_num = 0
    # we set chosen_num to a random number between 1 and 10
    chosen_num = str(random.randint(1, 10))
    print("Round " + str(rounds))
    print("Input a number between 1 and 10")
    # create a guess variable to store the user's guess
    guess = 0
    # use a while loop to let the user guess until they guess correctly
    while guess != chosen_num:
        # have the user enter in a number (which is returned as a string)
        guess = input()
        # check if the user's guess equals the chosen number
        if guess == chosen_num:
            print("Wow you guessed right!")
        else:
            print("Guess again...")
    # add 1 to rounds
    rounds = rounds + 1
# Thank the user for playing
print("Thanks for playing!")
